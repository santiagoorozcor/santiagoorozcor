        print(

            f"Role: {self.role}\n"
            f"Last Education: {self.last_education}\n"
            f"Industries Worked: {', '.join(self.industries_worked)}\n"
            f"Current learning tech tools: {', '.join(self.get_current_learning_goals_summary())}"
        )

        return [
            "AWS certification",

        ]
